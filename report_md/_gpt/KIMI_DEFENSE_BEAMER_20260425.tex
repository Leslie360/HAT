\section{Introduction}

\begin{frame}{The Analog CIM Opportunity}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Energy wall:} Data movement dominates AI inference; analog CIM offers 10--100$\times$ theoretical savings via Kirchhoff-current accumulation.
        \item \textbf{Organic optoelectronics:} Wafer-scale, low-temperature, flexible substrates with inherent photosensitivity---enabling sensor-compute fusion.
        \item \textbf{The catch:} Device variability, nonlinear write, and retention drift break deterministic digital assumptions.
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Key Numbers}
        \begin{itemize}
          \item Tiny-ViT-5M FP32: \textbf{98.06\%} (\dataset{CIFAR-10})
          \item Analog-mapped params: \textbf{$\sim$88\%}
          \item Energy reduction: \textbf{11.45$\times$} (first-order estimate)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Four Deployment Constraints}
  \begin{enumerate}
    \item \textbf{ADC resolution} --- below 6 bits causes abrupt collapse ($S_{\text{ADC}} = 0.98$)
    \item \textbf{Sublinear photoresponse} --- inverse-gamma compensation restores +5.8~pp at $\gamma_{\text{phys}} = 2.0$
    \item \textbf{Hardware-instance overfitting} --- Standard HAT collapses to 10.00\% on fresh arrays
    \item \textbf{Severe nonlinear write} --- NL=2.0 recovers to $\sim$80--82\% band with revised gradient scaling
  \end{enumerate}
  \vfill
  \centering
  \textbf{Central claim:} Ensemble HAT raises fresh-instance accuracy from 10.00\% to \textbf{86.37$\pm$1.54\%}.
\end{frame}

\section{Simulation Framework}

\begin{frame}{Hybrid Analog/Digital Stack}
  \begin{columns}
    \begin{column}{0.60\textwidth}
      \textbf{Analog partition:}
      \begin{itemize}
        \item Patch embedding, Q/K/V projections, output projections, MLP layers
        \item High weight reuse, regular access patterns
      \end{itemize}
      \textbf{Digital partition:}
      \begin{itemize}
        \item Softmax, LayerNorm, GELU, positional encoding, residual adds
        \item Dynamic, precision-critical operations
      \end{itemize}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{block}{Parameter Split}
        Analog: $\sim$88\%\\
        Digital: $\sim$12\%\\
        (Energy: digital $\sim$60\% due to attention)
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Profile-Driven Simulation}
  \textbf{Forward path:} Weight $\rightarrow$ differential conductance ($G^+$, $G^-$) $\rightarrow$ STE quantization $\rightarrow$ D2D mismatch + C2C noise $\rightarrow$ ADC $\rightarrow$ scale recovery
  \vfill
  \textbf{Profile interface (JSON):}
  \begin{itemize}
    \item $\sigma_{\text{D2D}}$, $\sigma_{\text{C2C}}$, $n_{\text{states}}$, $G_{\min}$, $G_{\max}$
    \item $NL_{\text{LTP}}$, $NL_{\text{LTD}}$, $\tau_1$, $\tau_2$, $A_0$
    \item $\gamma_{\text{phys}}$, $I_{\text{dark}}$, $\ldots$
  \end{itemize}
  \vfill
  \textbf{Key feature:} Swap the profile $\rightarrow$ change technology without changing code.
\end{frame}

\section{Ensemble HAT Methodology}

\begin{frame}{Standard HAT vs Ensemble HAT}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Standard HAT:}
      \begin{equation*}
        \theta^*_{\text{std}} = \arg\min_\theta \frac{1}{N}\sum_{n=1}^N \ell(f(x_n; \theta, M_0), y_n)
      \end{equation*}
      \begin{itemize}
        \item One fixed D2D mask $M_0$ throughout training
        \item Overfits to a specific hardware instance
        \item \textcolor{red}{Collapses to 10.00\% on fresh instances}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Ensemble HAT:}
      \begin{equation*}
        \theta^*_{\text{ens}} = \arg\min_\theta \mathbb{E}_{M \sim p(M)}\!\left[\frac{1}{N}\sum_{n=1}^N \ell(f(x_n; \theta, M), y_n)\right]
      \end{equation*}
      \begin{itemize}
        \item Resamples $M^{(e)}$ at each epoch
        \item Matches deployment distribution
        \item \textcolor{green!60!black}{86.37$\pm$1.54\% on fresh instances}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Theoretical Foundation}
  Second-order Taylor expansion of the Ensemble HAT objective reveals:
  \begin{equation*}
    \mathcal{L}_{\text{ens}} \approx \mathcal{L}_{\text{std}} + \frac{\sigma^2_{\text{D2D}}}{2} \sum_i \theta_i^2 \left(\frac{\partial \ell}{\partial \theta_i}\right)^2
  \end{equation*}
  \vfill
  \begin{itemize}
    \item \textbf{Implicit regularizer:} Fisher-weighted gradient-$L_2$ penalty
    \item \textbf{Structural analogue:} Stochastic SAM (Sharpness-Aware Minimization) along D2D direction
    \item \textbf{PAC-Bayes bound:} Generalization to fresh instances tracks training-distribution average
    \item \textbf{No extra hyperparameters:} Regularization strength set by physical $\sigma_{\text{D2D}}$
  \end{itemize}
\end{frame}

\section{Results}

\begin{frame}{Canonical Deployment Results}
  \begin{columns}
    \begin{column}{0.50\textwidth}
      \textbf{Canonical regime ($NL=1.0$):}
      \begin{itemize}
        \item Ensemble HAT: \textbf{86.37$\pm$1.54\%}
        \item Standard HAT fresh: \textbf{10.00\%}
        \item OPECT zero-shot: \textbf{88.53$\pm$0.08\%}
      \end{itemize}
      \textbf{Severe-NL regime ($NL=2.0$):}
      \begin{itemize}
        \item Standard uniform: \textbf{81.20\%}
        \item Ensemble uniform: \textbf{80.54\%}
        \item Ensemble proportional: \textbf{80.83\%}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Key Insights}
        \begin{itemize}
          \item All true-NL2 routes recover to $\sim$80--82\% band
          \item Standard vs Ensemble do not separate under NL=2.0
          \item Proportional noise offers no special advantage
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{ADC Resolution: The 6-Bit Cliff}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Sobol analysis:}
      \begin{itemize}
        \item Full grid: $S_{\text{ADC}} = 0.98$ (dominates)
        \item Operational envelope: $S_{\text{D2D}} = 0.92$ (dominates)
      \end{itemize}
      \textbf{Two-phase hierarchy:}
      \begin{enumerate}
        \item Secure $\geq$6-bit ADC first
        \item Then minimize D2D variability
      \end{enumerate}
      \vfill
      5-to-6-bit transition: consistent $\sim$7~pp jump across all D2D levels.
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{block}{Design Rule}
        \centering
        \textbf{6 bits = minimum viable}\\
        for transformer CIM
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Zero-Shot Transfer to Literature-Calibrated Device}
  \textbf{OPECT array (Zhang 2025):} $\sigma_{\text{D2D}} = 3\%$, $\sigma_{\text{C2C}} = 2\%$, 34 states, $G_{\max}/G_{\min} = 47.3$
  \vfill
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Standard HAT:}
      \begin{itemize}
        \item Collapses to 10.00\% across all profiles
        \item Fails on PCM, RRAM alternatives
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Ensemble HAT:}
      \begin{itemize}
        \item Reaches \textbf{88.53$\pm$0.08\%}
        \item No profile-specific retraining
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
  \centering
  \textbf{Same framework, different JSON profile.} Demonstrates profile-driven extensibility.
\end{frame}

\section{Limitations}

\begin{frame}{Limitations and Future Work}
  \begin{itemize}
    \item \textbf{Simulation level:} Behavioral, not circuit-accurate. IR drop, sneak paths, thermal effects are scalar placeholders.
    \item \textbf{Dataset scale:} CIFAR-10/100 and Flowers-102. ImageNet validation is future work.
    \item \textbf{Energy model:} First-order analytical with placeholder per-operation constants. Not silicon-validated.
    \item \textbf{Device profiles:} Literature-calibrated; measured-device calibration pending PhD collaborator data.
    \item \textbf{ADC modeling:} Hook diagnostics, not physical ADC boundary implementation.
  \end{itemize}
  \vfill
  \centering
  Framework purpose: \textbf{materials-to-system risk ranking before fabrication.}
\end{frame}

\section{Conclusion}

\begin{frame}{Summary}
  \begin{enumerate}
    \item \textbf{Framework:} First profile-driven behavioral simulation for organic optoelectronic CIM with ViT backbones.
    \item \textbf{Discovery:} Hardware-instance overfitting is the dominant deployment risk, not nominal quantization.
    \item \textbf{Solution:} Ensemble HAT resamples D2D masks per epoch, raising fresh-instance accuracy from 10.00\% to 86.37\%.
    \item \textbf{Theory:} Second-order expansion reveals implicit regularizer + SAM analogue + PAC-Bayes bound.
    \item \textbf{Design rules:} Secure $\geq$6-bit ADC, then control D2D, then use Ensemble HAT.
  \end{enumerate}
\end{frame}
